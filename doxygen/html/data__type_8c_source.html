31int bigint_to_hex(OUT char* str, IN bigint** src) {
32 if (*src == NULL) return -1; // NULL 체크
33
34 int idx = 0;
35 int buffer_size = 0;
36 const char* format = NULL; // 포맷 문자열 초기화
37
38 // WORD_BITS에 따른 형식과 버퍼 크기 설정
39 switch (WORD_BITS) {
40 case 64:
41 format = "%016" PRIx64;
42 buffer_size = 17;
43 break;
44 case 32:
45 format = "%08" PRIx32;
46 buffer_size = 9;
47 break;
48 case 16:
49 format = "%04" PRIx16;
50 buffer_size = 5;
51 break;
52 case 8:
53 format = "%02" PRIx8;
54 buffer_size = 3;
55 break;
56 default:
57 return -2; // Unsupported WORD_BITS
58 }
59
60 // 부호 처리
61 if ((*src)->sign == 1) {
62 str[idx++] = '-';
63 }
64
65 // 16진수 접두사 추가
66 str[idx++] = '0';
67 str[idx++] = 'x';
68
69 // 각 워드 값을 출력
70 for (int i = (*src)->word_len - 1; i >= 0; i--) {
71 idx += snprintf(&str[idx], buffer_size, format, (*src)->a[i]);
72 }
73
74 // 문자열 종료 처리
75 str[idx] = '\0';
76
77 return idx; // 최종 문자열 길이 반환
78}
