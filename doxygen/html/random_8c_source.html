82 ssize_t ret;
83
84 while (fd == -1){
85 fd = open("/dev/urandom", O_RDONLY);
86 if (fd == -1 && errno == EINTR)
87 continue;
88 else if (fd == -1)
89 return GEN_RANDOM_BYTES_FAIL;
90 }
91
92 while (byte_len > 0){
93 do{
94 ret = read(fd, dst, byte_len);
95 }while(dst[byte_len - 1] == 0);
96 if (ret == -1 && errno == EINTR)
97 continue;
98 else if (ret == -1)
99 return GEN_RANDOM_BYTES_FAIL;
100 dst += ret;
101 byte_len -= ret;
102 }
103
104 return GEN_RANDOM_BYTES_SUCCESS;
