12msg bi_add(OUT bigint** dst, IN bigint** a, IN bigint** b){
13 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
14 if((*a)->sign == 0 && (*b)->sign == 1){ // a - b
15 (*b)->sign = 0;
16 if(bi_sub(dst, a, b) != BI_SUB_SUCCESS) return BI_ADD_FAIL; // a - b
17 if(*dst != *b) (*b)->sign = 1;
18 return BI_ADD_SUCCESS;
19 }
20 if ((*a)->sign == 1 && (*b)->sign == 0){ // -a + b = b - a
21 (*a)->sign = 0;
22 if(bi_sub(dst, b, a) != BI_SUB_SUCCESS) return BI_SUB_FAIL; // b - a
23 if(*dst != *a) (*a)->sign = 1;
24 return BI_ADD_SUCCESS;
25 }
26 if ((*a)->sign == 1 && (*b)->sign == 1)
27 { // -a + -b = -(a + b)
28 (*a)->sign = 0;
29 (*b)->sign = 0;
30 if(bi_add(dst, a, b) != BI_ADD_SUCCESS) return BI_ADD_FAIL; // a + b
31 if(*dst != *a) (*a)->sign = 1;
32 if(*dst != *b) (*b)->sign = 1;
33 (*dst)->sign = 1; // -(a + b)
34 return BI_ADD_SUCCESS;
35 }
36
37 byte carry = 0;
38 int max_word_len = max((*a)->word_len, (*b)->word_len);
39 int dst_word_len = max_word_len + 1;
40 word temp_a = 0, temp_b = 0; // dst가 a와 b 중 같은 값일 경우 사용
41 int word_size_a = (*a)->word_len;
42 int word_size_b = (*b)->word_len;
43
44 if(*dst == NULL){
45 if(bi_new(dst, dst_word_len) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
46 }else if(*dst != NULL && (*dst)->word_len < max_word_len + 1){ // 해당 부분은 나중에 bi_resize 함수 만들어서 최적화 해보자
47 if(bi_resize(dst, dst_word_len) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
48 }
49 // dst가 a와 b가 아닌 경우 길이는 동일하게 위에서 체크 했으니까 재사용
50
51 // 두 빅넘 크기 맞추기
52 if(bi_expand(b, max_word_len, 0) != BI_EXPAND_SUCCESS) return BI_EXPAND_FAIL;
53 if(bi_expand(a, max_word_len, 0) != BI_EXPAND_SUCCESS) return BI_EXPAND_FAIL;
54
55 // 덧셈 연산 수행
56 for(int i = 0; i < max_word_len; i++){
57 temp_a = (*a)->a[i];
58 temp_b = (*b)->a[i];
59 (*dst)->a[i] = (word)(temp_a + temp_b + carry);
60 carry = (temp_a > MAX_VALUE - (temp_b + carry) || temp_b > MAX_VALUE - (temp_a + carry) || (temp_a == MAX_VALUE && temp_b == MAX_VALUE)) ? 1 : 0; // carry bit 계산
61 }
62 if (carry) (*dst)->a[max_word_len] = carry;
63 (*dst)->sign = 0;
64
65 if(bi_resize(dst, dst_word_len - !carry) != BI_RESIZE_SUCCESS){ // carry ^ 1을 한 이유는 마지막 carry가 발생하지 않으면 사이즈를 1 word 줄여도 되서
66 if(bi_delete(dst) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
67 return BI_RESIZE_FAIL;
68 }
69
70 if(*dst != *a && bi_resize(a, word_size_a) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
71 if(*dst != *b && bi_resize(b, word_size_b) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
72
73 return BI_ADD_SUCCESS;
74}

77* Name: bi_sub
78*
79* Description: Bigint substraction
80*
81* Arguments: - bigint** dst: pointer to bigint struct
82* - bigint** a: bigint struct
83* - bigint** b: bigint struct

85msg bi_sub(OUT bigint** dst, IN bigint** a, IN bigint** b){
86 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
87
88 if ((*a)->sign == 0 && (*b)->sign == 1){ // a - (-b) = a + b
89 (*b)->sign = 0;
90 if(bi_add(dst, a, b) != BI_ADD_SUCCESS) return BI_SUB_FAIL;
91 if(*dst != *b) (*b)->sign = 1;
92 return BI_SUB_SUCCESS;
93 }
94 else if ((*a)->sign == 1 && (*b)->sign == 0){ // -a - b = -(a + b)
95 (*a)->sign = 0;
96 if(bi_add(dst, a, b) != BI_ADD_SUCCESS) return BI_SUB_FAIL;
97 if(*dst != *a) (*a)->sign = 1;
98 (*dst)->sign = 1;
99 return BI_SUB_SUCCESS;
100 }
101 else if ((*a)->sign == 1 && (*b)->sign == 1){ // -a - (-b) = b - a
102 (*a)->sign = 0;
103 (*b)->sign = 0;
104 if(bi_sub(dst, b, a) != BI_SUB_SUCCESS) return BI_SUB_FAIL;
105 if(*dst != *a) (*a)->sign = 1;
106 if(*dst != *b) (*b)->sign = 1;
107 return BI_SUB_SUCCESS;
108 }
109
110 // b가 a보다 더 큰 경우 (a - b = -(b - a))
111 if(bi_compare_abs(a, b) == -1){
112 // b - a
113 if (bi_sub(dst, b, a) != BI_SUB_SUCCESS)
114 return BI_SUB_FAIL;
115 (*dst)->sign = 1; // -(b - a) // 모듈러 체크
116 return BI_SUB_SUCCESS;
117 }
118
119 int max_word_len = (*a)->word_len;
120 int word_size_b = (*b)->word_len;
121
122 // 여기서 첫 barrow 계산할 때 a랑 b를 전체를 넣으면 안되고 a[0] b[0]끼리 비교를 해줘야 한다.
123 byte borrow = (*a)->a[0] >= (*b)->a[0] ? 0 : 1; // 여기 첫 borrow 수행할 때 a >= b일 때 0, a < b 일 때 1
124 byte borrow_temp = 0; // 이전 word에서 사용한 borrow 값을 뺄 때 사용
125
126 if(*dst == NULL){
127 if(bi_new(dst, max_word_len) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
128 }else if(*dst != NULL && (*dst)->word_len < max_word_len){ // 해당 부분은 나중에 bi_resize 함수 만들어서 최적화 해보자
129 if(bi_resize(dst, max_word_len) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
130 }
131
132 if(bi_expand(b, max_word_len, 0) != BI_EXPAND_SUCCESS) return BI_EXPAND_FAIL;
133 // 뺄셈 연산 수행
134
135 for(int i = 0; i < (*a)->word_len - 1; i++){
136 // borrow를 사용할 경우 0x100000000을 더해줘야 하는데 이를 0xffffffff + 0x1로 나누어 더해 준다. 그리고 이전에 빌려갔던 borrow_temp를 빼준다.
137 (*dst)->a[i] = (word)(borrow * MAX_VALUE - (*b)->a[i] + (*a)->a[i] + borrow - borrow_temp);
138 borrow_temp = borrow;
139 borrow = ((*a)->a[i + 1] < (*b)->a[i + 1] + borrow || ((*b)->a[i + 1] == MAX_VALUE && borrow)) ? 1 : 0; // borrow bit 계산
140 }
141 (*dst)->a[max_word_len - 1] = (word)(borrow * MAX_VALUE - (*b)->a[max_word_len - 1] + (*a)->a[max_word_len - 1] + borrow - borrow_temp);
142
143 // 부호 설정
144 (*dst)->sign = 0;
145 if(bi_resize(dst, max_word_len) != BI_RESIZE_SUCCESS){
146 if(bi_delete(dst) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
147 return BI_RESIZE_FAIL;
148 }
149
150 if(*dst!= *b && bi_resize(b, word_size_b) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
151 return BI_SUB_SUCCESS;
152}

168/*************************************************
169* Name: bi_mul
170*
171* Description: Bigint multiplication
172*
173* Arguments: - bigint** dst: pointer to bigint struct
174* - bigint** a: bigint struct
175* - bigint** b: bigint struct
176**************************************************/
177msg bi_mul(OUT bigint **dst, IN bigint **a, IN bigint **b){
178 // 나중에 자료형 단위가 달라져도 동작할 수 있게 수정하기
179 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
180
181 msg result_msg = BI_MUL_FAIL;
182 if(bi_compare_abs(a, b) == -1){ // a < b
183 result_msg = bi_mul(dst, b, a);
184 if(result_msg != BI_MUL_SUCCESS) return result_msg;
185 return BI_MUL_SUCCESS;
186 }
187
188 bigint* temp1 = NULL;
189 bigint* temp2 = NULL;
190 bigint* dst_temp = NULL;
191 word b_i;
192 int temp_len_a = (*a)->word_len;
193 int temp_len_b = (*b)->word_len;
194 int idx_temp1, idx_temp2;
195 byte a_sign, b_sign;
196 int max_word_len = (*a)->word_len + (*b)->word_len;
197
198 if(bi_new(&dst_temp, max_word_len) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
199
200 result_msg = bi_new(&(temp1), temp_len_a);
201 if(result_msg != BI_ALLOC_SUCCESS) goto MUL_EXIT;
202 result_msg = bi_new(&(temp2), temp_len_a);
203 if(result_msg != BI_ALLOC_SUCCESS) goto MUL_EXIT;
204
205 a_sign = (*a)->sign;
206 b_sign = (*b)->sign;
207 (*a)->sign = 0;
208 (*b)->sign = 0;
209
210 for (int i = (temp_len_b * 2) - 1; i >= 0; i--){
211 b_i = (i % 2 == 0) ? GET_LOWER_PART((*b)->a[i / 2]) : GET_UPPER_PART((*b)->a[i / 2]);
212
213 idx_temp1 = 0;
214 idx_temp2 = 0;
215 // temp1, temp2 초기화
216 for(int j = 0; j < temp1->word_len; j++) temp1->a[j] = 0;
217 for(int j = 0; j < temp2->word_len; j++) temp2->a[j] = 0;
218
219 // 곱셈 수행
220 for (int j = 0; j < temp_len_a * 2; j++) { // word_len == 2
221 if (j % 2 == 0) temp1->a[idx_temp1++] = (word)(b_i * GET_LOWER_PART((*a)->a[j / 2]));
222 else temp2->a[idx_temp2++] = (word)(b_i * GET_UPPER_PART((*a)->a[j / 2]));
223 }
224
225 result_msg = bi_shift_left(&temp1, &temp1, i * WORD_SHIFT); // 16비트 단위로 밀기
226 if (result_msg != BI_SHIFT_SUCCESS)
227 goto MUL_EXIT;
228
229 result_msg = bi_shift_left(&temp2, &temp2, (i + 1) * WORD_SHIFT); // 16비트 단위로 밀기
230 if (result_msg != BI_SHIFT_SUCCESS)
231 goto MUL_EXIT;
232
233 result_msg = bi_add(&temp1, &temp1, &temp2); // 덧셈 수행
234 if(result_msg != BI_ADD_SUCCESS) goto MUL_EXIT;
235 result_msg = bi_add(&dst_temp, &dst_temp, &temp1); // 덧셈 수행
236 if(result_msg != BI_ADD_SUCCESS) goto MUL_EXIT;
237 }
238
239 // 부호 처리
240 dst_temp->sign = a_sign ^ b_sign; // XOR 연산으로 부호 처리 다르면 음수, 같으면 양수
241 (*a)->sign = a_sign;
242 (*b)->sign = b_sign;
243
244 if (bi_assign(dst, &dst_temp) != BI_SET_ASSIGN_SUCCESS) goto MUL_EXIT;
245 if (bi_resize(dst, max_word_len) != BI_RESIZE_SUCCESS) goto MUL_EXIT;
246 result_msg = BI_MUL_SUCCESS;
247
248MUL_EXIT:
249 if (bi_delete(&temp1) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
250 if (bi_delete(&temp2) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
251 return result_msg;
252}

254/*************************************************
255* Name: bi_mul_karachuba
256*
257* Description: Bigint multiplication using Karachuba algorithm
258*
259* Arguments: - bigint** dst: pointer to bigint struct
260* - bigint** a: bigint struct
261* - bigint** b: bigint struct
262**************************************************/
263// 카라츄바의 조건상 두 입력 값의 크기는 동일한 크기이다.
264msg bi_mul_karachuba(OUT bigint **dst, IN bigint **a, IN bigint **b, IN int karachuba_flag){
265 if (*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
266 msg result_msg = BI_MUL_FAIL;
267
268 // 입력 값이 동일한 조건으로 인해 아래 함수는 필요 없음 -> 입력 값이 다를 경우 필요
269// if(bi_refine(a) != BI_SET_REFINE_SUCCESS) return BI_SET_REFINE_FAIL;
270// if(bi_refine(b) != BI_SET_REFINE_SUCCESS) return BI_SET_REFINE_FAIL;
271
272// int word_len = (*a)->word_len; // 입력 값이 같기에 하나만 가져오면 된다.
273
274 int len_a = (*a)->word_len;
275 int len_b = (*b)->word_len;
276 int max_word_len = max(len_a, len_b); // 입력 값 중 최댓값 가져오기 -> 입력 값 크기가 다를 경우 필요
277 int min_word_len = min(len_a, len_b); // 입력 값 중 최솟값 가져오기 -> 입력 값 크기가 다를 경우 필요
278 int dst_word_len = len_a + len_b;
279
280 // base case에서 카라츄바가 아닌 일반 곱셈 수행을 위한 연산 -> 입력 값 크기가 다를 경우 word_len을 max_word_len으로 변경
281 if(karachuba_flag >= min_word_len || karachuba_flag <= 1){
282 result_msg = bi_mul(dst, a, b);
283 if(result_msg != BI_MUL_SUCCESS) return result_msg;
284 return BI_MUL_SUCCESS;
285 }
286
287 bigint* a_0 = g_pool.pool[g_pool.current_depth][0];
288 bigint* b_0 = g_pool.pool[g_pool.current_depth][1];
289 bigint* a_1 = g_pool.pool[g_pool.current_depth][2];
290 bigint* b_1 = g_pool.pool[g_pool.current_depth][3];
291 bigint* a_0b_0 = g_pool.pool[g_pool.current_depth][4];
292 bigint* a_1b_1 = g_pool.pool[g_pool.current_depth][5];
293 bigint* a_1_a_0 = g_pool.pool[g_pool.current_depth][6];
294 bigint* b_1_b_0 = g_pool.pool[g_pool.current_depth][7];
295 g_pool.current_depth++;
296
297 byte a_sign = 0, b_sign = 0;
298// int half_word_len = (word_len + 1) >> 1; // 입력 값 중 최댓값의 절반 값 -> 입력 값 크기가 같은 경우
299
300 int half_word_len = (max_word_len + 1) >> 1; // 입력 값 중 최댓값의 절반 값 -> 입력 값 크기가 다를 경우
301 // a와 b의 사이즈가 다른 경우 사이즈를 동일하게 맞춰줘야 함 -> 이것도 입력 값 크기가 다를 경우에만 사용
302 if((*a)->word_len != max_word_len){ // a < b
303 if(bi_resize(a, max_word_len) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
304 }else if((*b)->word_len != max_word_len){ // a > b
305 if(bi_resize(b, max_word_len) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
306 }
307
308 a_sign = (*a)->sign;
309 b_sign = (*b)->sign;
310 (*a)->sign = 0;
311 (*b)->sign = 0;
312
313 // A_1, B_1 계산
314 result_msg = bi_shift_right(&a_1, a, half_word_len * WORD_BITS);
315 if (result_msg != BI_SHIFT_SUCCESS) goto karachuba_exit;
316
317 result_msg = bi_shift_right(&b_1, b, half_word_len * WORD_BITS);
318 if (result_msg != BI_SHIFT_SUCCESS) goto karachuba_exit;
319
320 // A_0, B_0 계산
321 result_msg = bi_get_lower(&a_0, a, half_word_len * WORD_BITS);
322 if (result_msg != BI_GET_LOWER_SUCCESS) goto karachuba_exit;
323
324 result_msg = bi_get_lower(&b_0, b, half_word_len * WORD_BITS);
325 if (result_msg != BI_GET_LOWER_SUCCESS) goto karachuba_exit;
326
327 result_msg = bi_mul_karachuba(&a_0b_0, &a_0, &b_0, karachuba_flag);
328 if(result_msg != BI_MUL_SUCCESS) goto karachuba_exit;
329
330 result_msg = bi_mul_karachuba(&a_1b_1, &a_1, &b_1, karachuba_flag);
331 if (result_msg != BI_MUL_SUCCESS) goto karachuba_exit;
332
333 // (A_1 * B_1) || (A_0 * B_0) => dst
334 result_msg = bi_cat(dst, &a_1b_1, &a_0b_0);
335 if (result_msg != BI_CAT_SUCCESS) goto karachuba_exit;
336
337 // (A_1 * B_1) + (A_0 * B_0) - (A_1 - A_0) * (B_1 - B_0)
338 result_msg = bi_sub(&a_1_a_0, &a_1, &a_0); // A_1 - A_0 => a_1_a_0
339 if(result_msg != BI_SUB_SUCCESS) goto karachuba_exit;
340
341 result_msg = bi_sub(&b_1_b_0, &b_1, &b_0); // B_1 - B_0 => b_1_b_0
342 if (result_msg != BI_SUB_SUCCESS) goto karachuba_exit;
343
344 result_msg = bi_add(&a_1b_1, &a_1b_1, &a_0b_0); // (A_1 * B_1) + (A_0 * B_0) => a_1b_1
345 if (result_msg != BI_ADD_SUCCESS) goto karachuba_exit;
346
347 // (A_1 - A_0) * (B_1 - B_0) => a_1_a_0 // 분할 정복
348 result_msg = bi_mul_karachuba(&a_1_a_0, &a_1_a_0, &b_1_b_0, karachuba_flag);
349 if (result_msg != BI_MUL_SUCCESS) goto karachuba_exit;
350
351 // (A_1 * B_1) + (A_0 * B_0) - (A_1 - A_0) * (B_1 - B_0) => a_1b_1
352 result_msg = bi_sub(&a_1b_1, &a_1b_1, &a_1_a_0);
353 if (result_msg != BI_SUB_SUCCESS) goto karachuba_exit;
354
355 // ((A_1 * B_1) + (A_0 * B_0) - (A_1 - A_0) * (B_1 - B_0)) << half_word_len * WORD_BITS => a_1b_1
356 result_msg = bi_shift_left(&a_1b_1, &a_1b_1, half_word_len * WORD_BITS);
357 if (result_msg != BI_SHIFT_SUCCESS) goto karachuba_exit;
358
359 // (A_1 * B_1) + ((A_1 * B_1) + (A_0 * B_0) - (A_1 - A_0) * (B_1 - B_0)) + (A_0 * B_0)
360 result_msg = bi_add(dst, dst, &a_1b_1);
361 if (result_msg != BI_ADD_SUCCESS) goto karachuba_exit;
362
363 // 부호 처리
364 (*dst)->sign = a_sign ^ b_sign; // XOR 연산으로 부호 처리 다르면 음수, 같으면 양수
365 if(*dst != *a) (*a)->sign = a_sign;
366 if(*dst != *b) (*b)->sign = b_sign;
367
368 result_msg = bi_resize(dst, dst_word_len);
369 if(result_msg != BI_RESIZE_SUCCESS) goto karachuba_exit;
370 if(*dst != *a) result_msg = bi_resize(a, len_a);
371 if(result_msg != BI_RESIZE_SUCCESS) goto karachuba_exit;
372 if(*dst != *b) result_msg = bi_resize(b, len_b);
373 if(result_msg != BI_RESIZE_SUCCESS) goto karachuba_exit;
374
375 result_msg = BI_MUL_SUCCESS;
376
377karachuba_exit:
378 g_pool.current_depth--;
379 return result_msg;
380}

440 *
441 * Arguments: - bigint** q: pointer to bigint struct
442 * - bigint** r: pointer to bigint struct
443 * - bigint** a: bigint struct
444 * - bigint** b: bigint struct
445 **************************************************/
446msg bi_div(OUT bigint **q, OUT bigint **r, IN bigint **a, IN bigint **b, IN int option){
447 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
448 // 몫과 나머지가 같은 메모리를 가리키면 안됨
449 if(*q != NULL && *q == *r) return DIV_FAIL;
450
451 msg result_msg = BI_DIV_FAIL;
452 bigint* temp_r = NULL;
453
454 // b가 0이거나 할당이 안되었을 경우
455 result_msg = bi_is_zero(b);
456 if(result_msg == BI_NOT_USING || result_msg == BI_IS_ZERO){
457 return BI_DIV_BY_ZERO;
458 }
459
460 if((*b)->sign){ // b가 음수인 경우
461 if((*a)->sign){ // a, b 모두 음수인 경우
462 (*a)->sign = 0;
463 (*b)->sign = 0;
464 result_msg = bi_div(q, r, a, b, option);
465 if(result_msg != BI_DIV_SUCCESS) return result_msg;
466 if(*a != *q && *a != *r) (*a)->sign = 1;
467 if(*b != *q && *b != *r) (*b)->sign = 1;
468 (*r)->sign = 1;
469 result_msg = BI_DIV_SUCCESS;
470 goto EXIT_DIV_REFINE;
471 }
472 // a가 양수인 경우
473 (*b)->sign = 0; // b는 양수로
474 (*a)->sign = 1; // a는 음수로
475 result_msg = bi_div(q, r, a, b, option);
476 if(result_msg != BI_DIV_SUCCESS) return result_msg;
477 if(*a != *r && *a != *q) (*a)->sign = 0;
478 if(*b != *r && *b != *q) (*b)->sign = 1;
479 result_msg = BI_DIV_SUCCESS;
480 goto EXIT_DIV_REFINE;
481 }else if((*a)->sign){ // a가 음수, b가 양수인 경우
482 // 1 만들기
483 bigint* one = NULL;
484 bigint* temp_r = NULL;
485 if(bi_new(&one, 1) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
486 one->a[0] = 1;
487
488 (*a)->sign = 0;
489 result_msg = bi_div(q, &temp_r, a, b, option);
490 if(result_msg != BI_DIV_SUCCESS) return result_msg;
491 if(*a != temp_r && *a != *q) (*a)->sign = 1;
492
493 // q와 r 처리
494 if(bi_sub(r, b, &temp_r) != BI_SUB_SUCCESS) return BI_SUB_FAIL; // r = b - r 수행. 여기서 나눗셈 연산 과정에서 b가 변하면 안된다!
495
496 (*q)->sign = 0; // -q
497 if(bi_add(q, q, &one) != BI_ADD_SUCCESS) return BI_ADD_FAIL; // q = -q - 1 수행 // 일단 여기서 문제 발생
498 (*q)->sign = 1; // q = -q - 1
499 result_msg = BI_DIV_SUCCESS;
500 goto EXIT_DIV_REFINE;
501 }
502
503 // 해당 코드로 짤 경우 time attack 가능성이 있음
504 int bi_compare = bi_compare_abs(a, b);
505 if(bi_compare <= 0){ // a가 b보다 작거나 같은 경우
506 if(*q == NULL){
507 if(bi_new(q, 1) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
508 }else{
509 if(bi_resize(q, 1) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
510 }
511 (*q)->word_len = 1;
512 if(!bi_compare){ // a == b인 경우
513 (*q)->a[0] = 1;
514 (*q)->sign = ((*a)->sign ^ (*b)->sign) ? 1 : 0; // 부호가 다르면 음수, 같으면 양수
515 if(*r == NULL){
516 if(bi_new(r, 1) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
517 }else{
518 if(bi_resize(r, 1) != BI_RESIZE_SUCCESS) return BI_RESIZE_FAIL;
519 }
520 (*r)->word_len = 1;
521 (*r)->a[0] = 0;
522 }else{ // case a < b
523 (*q)->a[0] = (word)(*a)->sign;
524 (*q)->sign = (*a)->sign;
525 // r 메모리 할당
526 // a가 음수인 경우 r = b + a, a가 양수인 경우 r = a
527 if(bi_assign(r, a) != BI_SET_ASSIGN_SUCCESS) return BI_ALLOC_FAIL; // 어차피 *r == *a 인경우 assign에서 바로 return
528 if((*a)->sign){ // a가 음수인 경우
529 if(bi_add(r, r, b) != BI_ADD_SUCCESS) return BI_ADD_FAIL;
530 }
531 }
532 result_msg = BI_DIV_SUCCESS;
533 goto EXIT_DIV_REFINE;
534 }
535
536 // 나눗셈을 위한 메모리 할당
537 int max_q_size = (*a)->word_len; // b가 음수가 아닐 경우
538 bigint* temp_q = NULL;
539
540 if(bi_new(&temp_q, max_q_size) != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
541
542 // r의 여부 상관 없이 무조건 할당하는 것이 더 효율적일 듯. a, b랑 같을 때만 할 수 는 있는데 코드가 너무 더러워짐
543 result_msg = bi_new(&temp_r, (*b)->word_len + 1);
544 if(result_msg != BI_ALLOC_SUCCESS) goto EXIT_DIV;
545
546 // WORD LONG DIVISION
547 if(option == WORD_LONG_DIV){
548 // general division k value 찾기
549 int k = 0;
550 word temp = (*b)->a[(*b)->word_len - 1];
551 while(temp){
552 temp >>= 1;
553 k++;
554 }
555 k = WORD_BITS - k;
556 result_msg = divc_gener(&temp_q, &temp_r, a, b, k);
557 }else{ // BINARY DIVISION
558 result_msg = divc(&temp_q, &temp_r, a, b);
559 }
560 if(result_msg != DIVC_SUCCESS) goto EXIT_DIV;
561
562 result_msg = bi_assign(q, &temp_q);
563 if(result_msg != BI_SET_ASSIGN_SUCCESS) goto EXIT_DIV;
564
565 // r이 NULL이면 할당
566 result_msg = bi_refine(&temp_r);
567 if(result_msg != BI_SET_REFINE_SUCCESS) goto EXIT_DIV;
568 result_msg = bi_assign(r, &temp_r);
569 if(result_msg != BI_SET_ASSIGN_SUCCESS) goto EXIT_DIV;
570 result_msg = BI_DIV_SUCCESS;
571
572EXIT_DIV:
573 if(bi_delete(&temp_r) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
574 if(bi_delete(&temp_q) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
575EXIT_DIV_REFINE:
576 if(bi_refine(r) != BI_SET_REFINE_SUCCESS) return BI_SET_REFINE_FAIL;
577 if(bi_refine(q) != BI_SET_REFINE_SUCCESS) return BI_SET_REFINE_FAIL;
578 return result_msg;
579}

581/*************************************************
582 * Name: divc
583 *
584 * Description: Bigint division using binary division
585 *
586 * Arguments: - bigint** q: pointer to bigint struct
587 * - bigint** r: pointer to bigint struct
588 * - bigint** a: bigint struct
589 * - bigint** b: bigint struct
590 **************************************************/
591msg divc(OUT bigint** q, OUT bigint** r, IN bigint** a, IN bigint** b){
592 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
593 for(int i = (*a)->word_len - 1; i >= 0; i--){
594 for(int j = WORD_BITS - 1; j >= 0; j--){ // bit 단위의 곱셈을 수행하기에
595 // R <- R << 1 | (a[i] >> j & 1) -> 이 버전은 bit division 버전이다.
596 bi_shift_left(r, r, 1); // r <- r << 1
597 (*r)->a[0] |= (((*a)->a[i] >> j) & 1); // r <- r | (a[i] >> j & 1)
598 // if R >= b then R <- R - b, Q[i] <- 1
599 if(bi_compare_abs(r, b) >= 0){ // 여기서 r이 재활용 되지 않은 경우는 상관 없는데 재활용 하는 경우 쓰레기 값이 있는지 확인해볼 필요가 있다. 내부에서 refine 해주기는 한다.
600 if(bi_sub(r, r, b) != BI_SUB_SUCCESS) return DIVC_FAIL;
601 (*r)->a[(*r)->word_len - 1] = 0;
602 (*q)->a[i] |= 1 << j;
603 }
604 }
605 }
606 return DIVC_SUCCESS;
607}

679/*************************************************
680 * Name: divcc
681 *
682 * Description: Bigint division using word division
683 *
684 * Arguments: - bigint** q: pointer to bigint struct
685 * - bigint** r: pointer to bigint struct
686 * - bigint** a: bigint struct
687 * - bigint** b: bigint struct
688 **************************************************/
689msg divcc(OUT word* q, OUT bigint** r, IN bigint** a, IN bigint** b){
690 if(*a == NULL || *b == NULL) return MEM_NOT_ALLOC;
691 msg result_msg = DIVCC_FAIL;
692 int n = (*a)->word_len;
693 int m = (*b)->word_len;
694 bigint* temp_a = NULL;
695 bigint* temp_b = NULL;
696 int power_decom[WORD_BITS] = {0, };
697 int decom_size = 0;
698
699 if(n == m)
700 *q = (word)((*a)->a[m - 1] / (*b)->a[m - 1]); // q = a[n] / b[n]
701 if(n == m + 1){
702 if((*a)->a[m] == (*b)->a[m - 1])
703// (*q) = (word)((1ULL << WORD_BITS) - 1);
704 (*q) = ~(word)0;
705 else{
706 result_msg = bi_new(&temp_a, 2);
707 if(result_msg != BI_ALLOC_SUCCESS) goto EXIT_DIVCC;
708 temp_a->a[1] = (*a)->a[m];
709 temp_a->a[0] = (*a)->a[m - 1];
710 result_msg = two_word_long_div(q, &temp_a, (*b)->a[m - 1]);
711 if(result_msg != TWO_WORD_LONG_DIV_SUCCESS) goto EXIT_DIVCC;
712 }
713 }
714 result_msg = bi_new(&temp_a, (*b)->word_len);
715 if(result_msg != BI_ALLOC_SUCCESS) return BI_ALLOC_FAIL;
716 // B * Q
717 decom_size = get_power_decomposition(*q, power_decom); // Q에 대한 power decomposition
718 result_msg = bi_assign(&temp_b, b);
719 if(result_msg != BI_SET_ASSIGN_SUCCESS) goto EXIT_DIVCC;
720 if(power_decom[0]){
721 result_msg = bi_assign(&temp_a, &temp_b);
722 if(result_msg != BI_SET_ASSIGN_SUCCESS) goto EXIT_DIVCC;
723 }
724 for(int i = 1; i < decom_size; i++){
725 result_msg = bi_shift_left(&temp_b, &temp_b, 1);
726 bi_refine(&temp_b);
727 if(result_msg != BI_SHIFT_SUCCESS) goto EXIT_DIVCC;
728 if(power_decom[i] == 0) continue;
729 result_msg = bi_add(&temp_a, &temp_b, &temp_a); // 여기서 덧셈 문제 발생
730 }
731
732 // A - B * Q
733 result_msg = bi_sub(r, a, &temp_a);
734 if(result_msg != BI_SUB_SUCCESS) goto EXIT_DIVCC;
735
736 while((*r)->sign){
737 (*q)--;
738 result_msg = bi_add(r, r, b);
739 if(result_msg != BI_ADD_SUCCESS) goto EXIT_DIVCC;
740 }
741
742 result_msg = DIVCC_SUCCESS;
743EXIT_DIVCC:
744 if(bi_delete(&temp_a) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
745 if(bi_delete(&temp_b) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
746 return result_msg;
747}

749msg two_word_long_div(OUT word* q, IN bigint** a, IN word b){
750 if(*a == NULL) return MEM_NOT_ALLOC;
751 if(b == 0) return BI_DIV_BY_ZERO;
752 if((*a)->word_len > 2) return TWO_WORD_LONG_DIV_LENGTH_INVALID;
753
754 (*q) = 0;
755 word r_temp = (*a)->a[1]; // R <- A_1
756 for(int i = WORD_BITS - 1; i >= 0; i--){
757 if(r_temp >= (1ULL << (WORD_BITS - 1))){
758 (*q) |= 1ULL << i;
759 r_temp = (r_temp << 1) + (((*a)->a[0] >> i) & 1) - b; // R <- 2R + a_j - B
760 }else{
761 r_temp = (r_temp << 1) + (((*a)->a[0] >> i) & 1); // R <- 2R + a_j
762 if(r_temp >= b){
763 (*q) |= 1ULL << i;
764 r_temp -= b;
765 }
766 }
767 }
768
769 return TWO_WORD_LONG_DIV_SUCCESS;
770}

772/*************************************************
773 * Name: bi_squ
774 *
775 * Description: Bigint squaring
776 *
777 * Arguments: - bigint** dst: pointer to bigint struct
778 * - bigint** a: bigint struct to be squared
779 **************************************************/
780// 데이터 넣을 때 조금 더 최적화 할 수 있는 방법 생각해보기
781msg bi_squ(OUT bigint** dst, IN bigint** a){
782 if(*a == NULL) return MEM_NOT_ALLOC;
783 msg result_msg = BI_SQU_FAIL;
784 word T2 = 0; // Ai * Aj 결과 저장
785 word temp = 0; // Aj 저장
786 bigint* C1 = NULL; // 제곱 결과 저장
787 bigint* C2 = NULL; // Ai * Aj 홀수 내용 저장
788 bigint* C3 = NULL; // Ai * Aj 짝수 내용 저장
789 bigint* mid = NULL; // C1과 C2의 중간 결과 값 저장
790 int max_word_len = (*a)->word_len * 2;
791
792 result_msg = bi_new(&C1, max_word_len);
793 if(result_msg != BI_ALLOC_SUCCESS) return result_msg;
794 result_msg = bi_new(&C2, max_word_len);
795 if(result_msg != BI_ALLOC_SUCCESS) goto EXIT_SQU;
796 result_msg = bi_new(&C3, max_word_len);
797 if(result_msg != BI_ALLOC_SUCCESS) goto EXIT_SQU;
798 result_msg = bi_new(&mid, max_word_len);
799 if(result_msg != BI_ALLOC_SUCCESS) goto EXIT_SQU;
800
801 for(int j = 0; j < max_word_len; j++){
802 C3->a[j] = 0; // C3 초기화 -> 처음에 j 번째만 초기화 하면 나머지는 덮어써짐
803 if(j % 2 == 0) temp = GET_LOWER_PART((*a)->a[j / 2]);
804 else if(j % 2 == 1) temp = GET_UPPER_PART((*a)->a[j / 2]);
805 C1 -> a[j] = temp * temp;
806 for(int i = j + 1; i < max_word_len; i++){
807 if(i % 2 == 1) T2 = temp * GET_UPPER_PART((*a)->a[i / 2]); // 위에서 연산했던 temp 재사용
808 else if(i % 2 == 0) T2 = temp * GET_LOWER_PART((*a)->a[i / 2]);
809 if((i + j) % 2 == 0) C3 ->a[(i + j) / 2] = T2; // 짝수
810 else C2 -> a[(i + j) / 2] = T2; // 홀수
811 }
812 result_msg = bi_shift_left(&C2, &C2, WORD_SHIFT); // 16비트 단위로 밀기
813 if(result_msg != BI_SHIFT_SUCCESS) goto EXIT_SQU;
814 result_msg = bi_add(&mid, &mid, &C2);
815 if(result_msg != BI_ADD_SUCCESS) goto EXIT_SQU;
816 result_msg = bi_add(&mid, &mid, &C3);
817 if(result_msg != BI_ADD_SUCCESS) goto EXIT_SQU;
818 // C2의 경우 j와 사용 마지막 인덱스 + 1만 초기화 해주면 나머지는 덮어써진다.
819 C2->a[j] = 0;
820 C2->a[(j + max_word_len) / 2] = 0;
821 }
822
823 result_msg = bi_shift_left(dst, &mid, 1); // 2배 만들기
824 if(result_msg != BI_SHIFT_SUCCESS) goto EXIT_SQU;
825
826 result_msg = bi_add(dst, dst, &C1); // 제곱 더하기
827 if(result_msg != BI_ADD_SUCCESS) goto EXIT_SQU;
828
829 result_msg = bi_resize(dst, max_word_len);
830 if(result_msg != BI_RESIZE_SUCCESS) goto EXIT_SQU;
831
832 (*dst) -> sign = 0; // 부호 설정
833
834 result_msg = BI_SQU_SUCCESS;
835
836EXIT_SQU:
837 if(bi_delete(&C1) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
838 if(bi_delete(&C2) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
839 if(bi_delete(&C3) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
840 if(bi_delete(&mid) != BI_FREE_SUCCESS) return BI_FREE_FAIL;
841
842 return result_msg;
843}

845/*************************************************
846 * Name: bi_squ_karachuba
847 *
848 * Description: Bigint squaring by using karachuba algorithm
849 *
850 * Arguments: - bigint** dst: pointer to bigint struct
851 * - bigint** a: bigint struct to be squared
852 **************************************************/
853msg bi_squ_karachuba(OUT bigint** dst, IN bigint** a, IN int squ_karachuba_flag){
854 if (*a == NULL) return MEM_NOT_ALLOC;
855 msg result_msg = 0;
856
857 // base case에서 카라츄바가 아닌 일반 제곱 수행을 위한 연산
858 if(squ_karachuba_flag >= (*a)->word_len || squ_karachuba_flag <= 1){
859 result_msg = bi_squ(dst, a);
860 if(result_msg != BI_SQU_SUCCESS) return result_msg;
861 return BI_SQU_SUCCESS;
862 }
863
864 bigint* a_0 = g_pool.pool[g_pool.current_depth][0];
865 bigint* a_1 = g_pool.pool[g_pool.current_depth][1];
866 bigint* a_1a_1 = g_pool.pool[g_pool.current_depth][2];
867 bigint* a_0a_0 = g_pool.pool[g_pool.current_depth][3];
868 g_pool.current_depth++;
869
870 (*a)->sign = 0;
871
872 int half_word_len = (((*a)->word_len) + 1) >> 1; // 길이의 절반 가져오기
873 int loop_len = half_word_len - ((*a)->word_len % 2); // 반복 횟수
874
875 if(bi_resize(&a_1, half_word_len) != BI_RESIZE_SUCCESS) return BI_SHIFT_FAIL;
876 if(bi_resize(&a_0, half_word_len) != BI_RESIZE_SUCCESS) return BI_SHIFT_FAIL;
877
878 // a_1, a_0 나누기
879 for(int i= 0; i < loop_len; i++){
880 a_1->a[i] = (*a)->a[i + half_word_len];
881 a_0->a[i] = (*a)->a[i];
882 }
883 if((*a)->word_len % 2 == 1) a_0->a[loop_len] = (*a)->a[loop_len]; // 길이가 홀수인 경우 반으로 나눌 때
884
885 result_msg = bi_squ_karachuba(&a_1a_1, &a_1, squ_karachuba_flag);
886 if(result_msg != BI_SQU_SUCCESS) goto karachuba_exit;
887
888 result_msg = bi_squ_karachuba(&a_0a_0, &a_0, squ_karachuba_flag);
889 if (result_msg != BI_SQU_SUCCESS) goto karachuba_exit;
890
891 // (A_1 * A_1) || (A_0 * A_0) => dst
892 result_msg = bi_cat(dst, &a_1a_1, &a_0a_0);
893 if (result_msg != BI_CAT_SUCCESS) goto karachuba_exit;
894
895 // A_1 * A_0 분할 정복
896 result_msg = bi_mul_karachuba(&a_0a_0, &a_1, &a_0, a_1->word_len / mul_karachuba_ratio);
897 if (result_msg != BI_MUL_SUCCESS) goto karachuba_exit;
898
899 // ((A_1 * A_0) << half_word_len * WORD_BITS + 1 => 2 * a_1a_0
900 result_msg = bi_shift_left(&a_0a_0, &a_0a_0, half_word_len * WORD_BITS + 1);
901 if (result_msg != BI_SHIFT_SUCCESS) goto karachuba_exit;
902
903 // (A_1 * A_1) + 2 * A_1 * A_0 + (A_0 * A_0)
