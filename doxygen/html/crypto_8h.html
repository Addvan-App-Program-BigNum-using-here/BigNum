160 result_msg = bi_sub(&q_minus_1, &q, &one);
161 if(result_msg != BI_SUB_SUCCESS) goto EXIT_RSA_KEYGEN;
162
163 // phi_n 계산
164 result_msg = bi_mul_karachuba(&phi_n, &p_minus_1, &q_minus_1, p_minus_1->word_len / mul_karachuba_ratio);
165 if(result_msg != BI_MUL_SUCCESS) goto EXIT_RSA_KEYGEN;
166
167 // n 생성
168 result_msg = bi_mul_karachuba(n, &p, &q, p->word_len / mul_karachuba_ratio);
169 if(result_msg != BI_MUL_SUCCESS) goto EXIT_RSA_KEYGEN;
170
171 // e 생성
172 result_msg = bi_set_from_string(e, "10001", 16);
173 if(result_msg != BI_SET_STRING_SUCCESS) goto EXIT_RSA_KEYGEN;
174
175 // 생성된 e가 gcd(phi_n, e) = 1을 만족해야 함. 현재 e는 10001로 소수이므로 만족, 나중에 임의의 p, q에 대해도 만족하도록 수정
176 result_msg = bi_gcd(&gcd, e, &phi_n);
177 if(result_msg != BI_GCD_SUCCESS) goto EXIT_RSA_KEYGEN;
178
179 result_msg = bi_compare(&gcd, &one);
180 if(result_msg != 0) goto EXIT_RSA_KEYGEN;
181 // 여기에 나중에 만족 안할 경우 로직 구현, 현재는 만족하는 경우만 구현
182
183 // d 생성
184 result_msg = bi_EEA(&gcd, &temp, d, &phi_n, e);
185 if(result_msg != BI_EEA_SUCCESS) goto EXIT_RSA_KEYGEN;
186 if((*d)->sign){
187 result_msg = bi_add(d, d, &phi_n);
188 if(result_msg != BI_ADD_SUCCESS) goto EXIT_RSA_KEYGEN;
189 }
190
191 result_msg = RSA_KEYGEN_SUCCESS;
192
193EXIT_RSA_KEYGEN:
194 bi_delete(&p);
195 bi_delete(&q);
196 bi_delete(&phi_n);
197 bi_delete(&one);
198 bi_delete(&p_minus_1);
199 bi_delete(&q_minus_1);
200 bi_delete(&temp);
201 bi_delete(&gcd);
202
203 return result_msg;
204}
